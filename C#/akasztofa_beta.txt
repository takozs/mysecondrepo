using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApplication2
{
    class Program
    {
        static int szorzat(int szam1, int szam2)
        {
            return szam1 * szam2;
        }
        static void kiiras()
        {
            Console.Write("ÁMSZKI");
            Console.Write("isn't");
            Console.Write("an iskola");
            Console.WriteLine();
        }
        static void vonal(string szoveg)
        {
            for (int i = 0; i < szoveg.Length; i++)
            {
                if (szoveg[i]!=' ')
                Console.Write("_");
                else Console.Write(" ");
            }
            Console.WriteLine();
        }
        
        static void Main(string[] args)
        {
            string szoveg = "ÁMSZKI nem iskola";
            char[] arnyek = new char[szoveg.Length];
            for (int i = 0; i < arnyek.Length; i++)
            {
                if (szoveg[i] != ' ') arnyek[i] = '_';
            }

            vonal(szoveg);
            bool vege;
            do
            {
                Console.WriteLine("Kérek egy betût!");
                char betu = Convert.ToChar(Console.ReadLine());

                for (int i = 0; i < szoveg.Length; i++)
                {
                    if (szoveg[i] == betu) arnyek[i] = betu;
                    Console.Write(arnyek[i]);
                }
                Console.WriteLine();
                vege = true;
                for (int i = 0; i < szoveg.Length; i++)
                {
                    if (szoveg[i] != arnyek[i]) vege = false;
                } 
            } while (vege=true);
            Console.WriteLine("Kitaláltad!");

            Console.ReadKey();
        }
    }
}
